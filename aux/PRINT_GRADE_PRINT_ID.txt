;PRINT GRADES & IDS SEPARATELY

;PRINT GRADES IN ORDER 
PRINT:
    CMP CX,SI
    JZ  EXIT
    MOV AL,GRADES[SI]
    AAM
    MOV BX,AX
    MOV DL,BH
    ADD DL,30H
    MOV AH,2
    INT 21H
    
    MOV DL,BL
    ADD DL,30H
    INT 21H
    
    MOV DL, 09H ; SPACE: 20H , TAB:09H
    INT 21H
    
    INC SI
    JMP PRINT    

  
EXIT: 
    

;PRINT IDS
MOV   CX , 100 ;4*N = 4*25= 100   
MOV   SI , 00 ; AS COUNTER


MOV DL, 0AH ; NEW LINE
MOV AH,2
INT 21H

SUBLOOP:
MOV   AX,IDS[SI] ; LAST 2-DIGITS OF ID
MOV   DX,AX
MOV   AH,2
INT   21H
XCHG  DL,DH
INT   21H
ADD   SI,2 

MOV   AX,IDS[SI] 
MOV   DX,AX
MOV   AH,2
INT   21H
XCHG  DL,DH
INT   21H
ADD   SI,2

MOV DL, 09H ; SPACE: 20H , TAB:09H
INT 21H
   
CMP   CX,SI
JA    SUBLOOP  ; END OF PRININT IDS
